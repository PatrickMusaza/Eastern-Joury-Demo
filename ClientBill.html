function loadAbbreviationsPage() { // Hide all tab contents const tabContents = document.querySelectorAll(".tab-content"); tabContents.forEach((content) => { content.style.display = "none"; }); // Show the Abbreviations tab content document.getElementById("abbreviationsContent").style.display = "block"; // Deactivate all nav links const navLinks = document.querySelectorAll(".nav-link"); navLinks.forEach((link) => { link.classList.remove("active"); }); // Activate the Abbreviations nav link document.getElementById("abbreviationsBtn").classList.add("active"); }
